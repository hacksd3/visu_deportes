           var equipos_ordenados = dataset.sort(function(a, b) {  
             return d3.ascending(a.equipo, b.equipo);

           function visualize() {  
                var table = d3.select("#principal")  
                     .append("table")  
                     .style("border-collapse", "collapse")  
                  .style("border", "2px black solid")  
                     .append("thead")  
                     .append("tbody");  
                var titulos = ["Equipos", "logo" ];  
                table.selectAll("th")  
                     .data(titulos)  
                     .enter()  
                     .append("th")  
                     .style("width", "100px")  
                     .text(function(titulos) {   
                          return titulos;   
                     });  
                var trs = table.selectAll("tr")  
                     .data(dataset).enter()  
                     .append("tr");  
                var equipoTd = trs.append("td")  
                     .text(function(d) {  
                          return d.equipo;  
                });  
                var logoTd = trs.append("td")  
                     .append("p")  
                     .attr("align", "center")  
                     .append("img")  
                          .attr("src", function(d) {  
                               return d.logo;  
                          })  
                          .attr("width", "20px")  
                          .attr("width", "20px")  
                          .attr("alt", function(d){  
                               return d.name;  
                          });  
           }
